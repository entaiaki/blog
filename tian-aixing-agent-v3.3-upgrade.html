            tag.contains("think", true)  || tag.contains("思考") -> THINKING
            // ... 中英文模糊匹配

    }
}

// EmotionParser.kt — 两阶段提取
fun parse(raw: String): EmotionResult {
    // 1. 正则提取 LLM 显式标记
    val tag = regex.find(raw)?.groupValues?.get(1)
    val emotion = if (tag != null) CharacterEmotion.fromTag(tag)
                  else guessEmotion(raw)  // 2. 关键词兜底
    return EmotionResult(raw.replace(regex, ""), emotion)
}

三、Kokoro TTS 本地 ONNX 推理

用 onnxruntime-android 在设备本地运行 Kokoro-82M TTS 模型， 零网络延迟，完全离线。

模型文件（assets/kokoro/）

文件大小说明

// 核心推理链
fun synthesize(text: String, voiceName: String): FloatArray {
    val tokens = textToTokens(text)
    val voice  = loadVoice(voiceName)
    val input  = OrtTensor(voice + tokens + velocity + prosody)
    val result = session.run(mapOf("input" to input))
    return result["audio"]!!.floatArray  // 24000 Hz, MONO, PCM16
}

KokoroTTSManager 队列播放

fun speak(text: String) {
    val sentences = splitSentences(text)
    // "你好！今天天气真好。" → ["你好！", "今天天气真好。"]
    val queue = ArrayDeque(sentences)
    playQueue()
}

private fun playQueue() {
    if (queue.isEmpty()) { _isTalking.value = false; return }
    val sentence = queue.removeFirst()
    _state.value = State.Synthesizing(sentence)
    val wav = kokoro.synthesizeToFile(sentence)
    _state.value = State.Playing(sentence)
    playWav(wav) { playQueue() }  // 递归下一句
}

Asset 加载技巧

从 Android assets 加载 ONNX 模型不能直接用路径，需要先用 context.assets.open() 读出字节数组，再 OrtSession.create(env, byteArray, sessionOptions)。 对于一个 86MB 的模型，读取和解析约需 2-3 秒。 

// LightAgentApplication.kt — App 启动时后台预加载
class LightAgentApplication : Application() {
    override fun onCreate() {
        super.onCreate()
        CoroutineScope(Dispatchers.IO).launch {
            KokoroTTSManager.getInstance(this@LightAgentApplication).initialize()
        }
    }
}

四、ChatViewModel 架构重构

TTS 从不归 ViewModel 管的独立组件，变成 ViewModel 的内置能力。 这消除了 ChatScreen 端的重复 TTS 初始化和消息监听。 

class ChatViewModel(application: Application) : AndroidViewModel(application) {
    // 新增
    private val ttsManager = KokoroTTSManager.getInstance(application)
    val isTalking: StateFlow<Boolean> = ttsManager.isTalking

    init {
        viewModelScope.launch(Dispatchers.IO) {
            ttsManager.initialize()  // 后台加载模型
        }
    }

    private fun speakReply(text: String) {
        try { ttsManager.speak(text) }
        catch (e: Exception) { Log.w("ChatVM", "TTS failed", e) }
    }

    override fun onCleared() {
        super.onCleared()
        ttsManager.release()
    }
}

消息发送流程：send() → LLM → EmotionParser → DesktopAgentService.updateEmotion() → speakReply()。ChatScreen 不再直接接触 TTS 组件。

踩坑记录

  • HuggingFace 文件名 ≠ 用户假设： 实际文件是 model_q8f16.onnx（非 model_q8.onnx）、分语音的单 .bin 文件（非单个 voices.bin）、 tokenizer.json（非 tokens.txt）。集成第三方模型前务必用 /tree/main 确认实际文件名。
  • LocalContext.current 不可在 remember 内调用： remember { TTSController(LocalContext.current) } 编译报错。 必须先在 Composable 作用域 val ctx = LocalContext.current 再传入。
  • ONNX Runtime Android 不支持路径加载 assets： 必须 context.assets.open(file).readBytes() → OrtSession.create(env, byteArray, opts)。
  • 悬浮窗权限需在 onResume 检查： 用户从设置页返回后不会触发 LaunchedEffect 重新执行，只有 onResume 才可靠。
  • Kokoro audio 输出是 PCM Float 数组（24000Hz, mono）： 直接写 AudioTrack 或用 WAV header 包裹后 MediaPlayer 播放均可。 当前方案使用 WAV 文件 + MediaPlayer（Manager 层），避免 AudioTrack 的缓冲区管理问题。
